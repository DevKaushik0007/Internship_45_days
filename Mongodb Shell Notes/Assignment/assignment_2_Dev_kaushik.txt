1) Group users by city and find the average number of users per role in each city.
--> [
  {
    $group: {
      _id: { city: "$city", role: "$role" },
      roleCount: { $sum: 1 },
    },
  },
  {
    $group: {
      _id: "$_id.city",
      totalRoles: { $sum: 1 },
      totalUsers: { $sum: "$roleCount" },
    },
  },
  {
    $project: {
      _id: 0,
      city: "$_id",
      avgUsersPerRole: {
        $divide: ["$totalUsers", "$totalRoles"],
      },
    },
  },
]
